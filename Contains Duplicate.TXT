public class Solution {
    public bool ContainsDuplicate(int[] nums) {
      
           HashSet<int>s = new HashSet<int>();
            for(int i = 0;i<nums.Length;i++)
            {
                if (s.Contains(nums[i]))
                    return true;
                s.Add(nums[i]);              
            }
            return false;
        
    }
}
